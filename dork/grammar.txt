[statement] -> [command] | [imperative] | [action]
[command] -> [imperative] [target] | [action] [target] | [option]
[imperative] -> [movement] | "open" | "take" | "drop" | "use"           /// an imperative is any command that alters the state of the game
[action] -> "look"          /// an action is any command that does not affect the state of the game
[option] -> "help" [imperative] | "help" [action] | "save" | "load" | "quit"            /// an option is any command that deals with the operation of the game
[target] -> [item] | [player] | "room"
